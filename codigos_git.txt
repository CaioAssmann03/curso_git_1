git init                 # Inicializa um repositório Git vazio
git add                  # Adiciona arquivos à área de stage
git commit -m            # Salva alterações com uma mensagem
git status               # Mostra o estado atual dos arquivos
git push -u origin master # Envia commits para o repositório remoto
git config name          # Define o nome do usuário do Git
git config email         # Define o e-mail do usuário do Git
git checkout             # Muda de branch ou restaura arquivos
git branch               # Lista, cria ou exclui branches

git ls-remote --symref [URL] HEAD       # Mostra a branch padrão do repositório remoto
git clone -b master [URL]               # Clona apenas a branch master
git clone [URL]                         # Clona o repositório (branch padrão)
cd [pasta]                              # Entra na pasta do projeto
git checkout master                     # Alterna para a branch master
git remote -v                           # Mostra os repositórios remotos configurados
ls -la                                  # Lista todos os arquivos (incluindo ocultos)
git ls-remote --heads origin            # Lista todas as branches do repositório remoto
git rm                      # Remove arquivos do Git e do sistema de arquivos  
git merge                # Combina branches (faz a junção das alterações)
git fetch                # Busca atualizações do remoto sem alterar seus arquivos
git reset                # Desfaz commits ou alterações (cuidado ao usar)
git stash                # Guarda temporariamente alterações não commitadas
git remote               # Mostra ou gerencia repositórios remotos configurados
git diff                 # Mostra diferenças entre arquivos/commits/branches
git log                      # Mostra histórico de commits (q para sair)
git log --oneline            # Histórico compacto (1 commit por linha)
git log -p                   # Mostra alterações em cada commit
git log --graph              # Exibe branches e merges visualmente
git show <commit-hash>       # Detalhes de um commit específico
git reflog                   # Mostra TODOS os movimentos (incluindo resets)

git rev-parse --show-toplevel  # Mostra o caminho completo da raiz do repositório Git atual
pwd                           # Mostra o caminho completo do diretório atual (comando do sistema)

git reset HEAD c.txt        # Remove do stage de exclusão
git checkout -- c.txt       # Restaura o arquivo

mv css/bannerinicia.css css/Banner_Inicia.css  # Renomeia o arquivo no sistema de arquivos (comando do sistema)
git rm css/bannerinicia.css                   # Remove o arquivo antigo do rastreamento do Git
git add css/Banner_Inicia.css                 # Adiciona o novo nome ao staging area do Git
